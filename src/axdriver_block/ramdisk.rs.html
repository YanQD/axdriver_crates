1//! Mock block devices that store data in RAM.
2
3extern crate alloc;
4
5use crate::BlockDriverOps;
6use alloc::{vec, vec::Vec};
7use axdriver_base::{BaseDriverOps, DevError, DevResult, DeviceType};
8
9const BLOCK_SIZE: usize = 512;
10
11/// A RAM disk that stores data in a vector.
12#[derive(Default)]
13pub struct RamDisk {
14    size: usize,
15    data: Vec<u8>,
16}
17
18impl RamDisk {
19    /// Creates a new RAM disk with the given size hint.
20    ///
21    /// The actual size of the RAM disk will be aligned upwards to the block
22    /// size (512 bytes).
23    pub fn new(size_hint: usize) -> Self {
24        let size = align_up(size_hint);
25        Self {
26            size,
27            data: vec![0; size],
28        }
29    }
30
31    /// Creates a new RAM disk from the exiting data.
32    ///
33    /// The actual size of the RAM disk will be aligned upwards to the block
34    /// size (512 bytes).
35    pub fn from(buf: &[u8]) -> Self {
36        let size = align_up(buf.len());
37        let mut data = vec![0; size];
38        data[..buf.len()].copy_from_slice(buf);
39        Self { size, data }
40    }
41
42    /// Returns the size of the RAM disk in bytes.
43    pub const fn size(&self) -> usize {
44        self.size
45    }
46}
47
48impl BaseDriverOps for RamDisk {
49    fn device_type(&self) -> DeviceType {
50        DeviceType::Block
51    }
52
53    fn device_name(&self) -> &str {
54        "ramdisk"
55    }
56}
57
58impl BlockDriverOps for RamDisk {
59    #[inline]
60    fn num_blocks(&self) -> u64 {
61        (self.size / BLOCK_SIZE) as u64
62    }
63
64    #[inline]
65    fn block_size(&self) -> usize {
66        BLOCK_SIZE
67    }
68
69    fn read_block(&mut self, block_id: u64, buf: &mut [u8]) -> DevResult {
70        let offset = block_id as usize * BLOCK_SIZE;
71        if offset + buf.len() > self.size {
72            return Err(DevError::Io);
73        }
74        if buf.len() % BLOCK_SIZE != 0 {
75            return Err(DevError::InvalidParam);
76        }
77        buf.copy_from_slice(&self.data[offset..offset + buf.len()]);
78        Ok(())
79    }
80
81    fn write_block(&mut self, block_id: u64, buf: &[u8]) -> DevResult {
82        let offset = block_id as usize * BLOCK_SIZE;
83        if offset + buf.len() > self.size {
84            return Err(DevError::Io);
85        }
86        if buf.len() % BLOCK_SIZE != 0 {
87            return Err(DevError::InvalidParam);
88        }
89        self.data[offset..offset + buf.len()].copy_from_slice(buf);
90        Ok(())
91    }
92
93    fn flush(&mut self) -> DevResult {
94        Ok(())
95    }
96}
97
98const fn align_up(val: usize) -> usize {
99    (val + BLOCK_SIZE - 1) & !(BLOCK_SIZE - 1)
100}
